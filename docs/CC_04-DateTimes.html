How much longer until the days get longer ??

def days_until_winter_solstice():
    today = date.today()
    current_year = today.year

    # Winter solstice is usually on Dec 21
    solstice = date(current_year, 12, 21)

    # If already past this year's solstice, compute for next year
    if today > solstice:
        solstice = date(current_year + 1, 12, 21)

    days_left = (solstice - today).days
    print(f"Only {days_left} days left until the shortest day of the year!")
    return days_left

# Example:
days_left = days_until_winter_solstice()
