9  Target Coding

Here is a small example, on how to implement target coding using just pandas

import numpy as np
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.metrics import r2_score

np.random.seed(42)

# -----------------------------
# 1. Small synthetic dataset
# -----------------------------
data = {
    "num1": np.random.normal(size=20),
    "num2": np.random.normal(size=20),
    "cat": ['A', 'B', 'C', 'A', np.nan, 'B', 'C', 'D', 'D', np.nan,
            'A', 'C', 'B', 'D', 'E', np.nan, 'E', 'B', 'C', 'A'],
    "y": np.random.normal(size=20)
}

df = pd.DataFrame(data)

X = df[["num1", "num2", "cat"]]
y = df["y"]
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=0)

# -----------------------------
# 2. Impute missing categorical values
# -----------------------------
X_train["cat"] = X_train["cat"].fillna("__MISSING__")
X_test["cat"] = X_test["cat"].fillna("__MISSING__")

# -----------------------------
# 3. Manual target encoding without map/transform/join
# -----------------------------
# Compute training category means
category_means = {}
for cat in X_train["cat"].unique():
    category_means[cat] = y_train[X_train["cat"] == cat].mean()

# Create target-encoded column in training set
cat_te_train = []
for val in X_train["cat"]:
    cat_te_train.append(category_means[val])
X_train["cat_te"] = cat_te_train

## When is Target Encoding equivalent to Dummy Coding ?
Below is a clean statsmodels example showing that:
1. With no other predictors, a regression with
* dummy coding, and * target encoding gives identical predictions.
2. With additional correlated predictors, this equivalence breaks down because the group means are no longer the OLS coefficients for the categorical variable.

✅ Statsmodels Example

import pandas as pd
import numpy as np
import statsmodels.api as sm

# -----------------------
# Example data
# -----------------------
df = pd.DataFrame({
    "group": ["A", "A", "A", "B", "B", "C", "C", "C", "C"],
    "y":     [10, 12, 11, 20, 22, 30, 31, 29, 32]
})

# -----------------------
# 1. Dummy-coded regression
# -----------------------
df_dum = pd.get_dummies(df["group"], drop_first=True).astype(float)
# now df_dum is numeric

X_dum = sm.add_constant(df_dum.astype(float))
y = df["y"].astype(float)

model_dummy = sm.OLS(y, X_dum).fit()
pred_dummy = model_dummy.predict(X_dum)

print("Dummy coding predictions:", pred_dummy.values)

# -----------------------
# 2. Target encoding regression
# -----------------------
means = df.groupby("group")["y"].mean()
df["target_enc"] = df["group"].map(means).astype(float)

X_te = sm.add_constant(df[["target_enc"]])
model_te = sm.OLS(y, X_te).fit()
pred_te = model_te.predict(X_te)

print("\nTarget encoding predictions:", pred_te.values)

print("\nIdentical predictions?", np.allclose(pred_dummy, pred_te))

Target encoding predictions: [11.  11.  11.  21.  21.  30.5 30.5 30.5 30.5]

Identical predictions? True

✔️ Output

np.allclose(...) → True The predictions match exactly.

Why? Because with only a categorical variable, the OLS fit is literally the group means, and target encoding plugs those straight in.

10 ❗ When do target encoding and dummy coding stop being equivalent?

The instant you introduce any other predictor that is correlated with the categorical variable, the equivalence breaks.

Example: Add a correlated predictor x.

df["x"] = [1,2,1,5,6,10,9,11,12]  # correlated with group

Then fit:

# Dummy coding
X_dum2 = sm.add_constant(pd.concat([df_dum, df["x"]], axis=1))
m1 = sm.OLS(df["y"], X_dum2).fit()
pred1 = m1.predict(X_dum2)

# Target encoding
X_te2 = sm.add_constant(pd.concat([df["target_enc"], df["x"]], axis=1))
m2 = sm.OLS(df["y"], X_te2).fit()
pred2 = m2.predict(X_te2)

print(np.allclose(pred1, pred2))

This will print:

False

✔️ Why the equivalence breaks

  • Dummy coding estimates conditional effects of category membership after adjusting for x.
  • Target encoding uses the marginal mean per category, not the conditional mean.
  • When x and group are correlated, marginal means are no longer the adjusted group effects.
  • So target encoding becomes a mis-specified transformation of the categorical variable.

In short:

Target encoding = correct only when the categorical predictor has no other correlated covariates in the model (or when the encoded mean is computed within cross-validated folds, but even then it is not equal to dummy coding).

Target Coding Leakage

Data Leakage can be especially severe in target coding for the case of rare categories.

import numpy as np
import pandas as pd
from sklearn.model_selection import train_test_split, KFold, cross_val_score
from sklearn.linear_model import LinearRegression
from sklearn.metrics import r2_score

np.random.seed(0)

# --------------------------
# 1. Create synthetic data
# --------------------------

n_categories = 200     # many categories
samples_per_cat = 3    # extremely rare
N = n_categories * samples_per_cat

categories = np.repeat(np.arange(n_categories), samples_per_cat)
np.random.shuffle(categories)

# Regression target: PURE NOISE
y = np.random.normal(loc=0.0, scale=1.0, size=N)

df = pd.DataFrame({"cat": categories, "y": y})

global_mean = df["y"].mean()

# ----------------------------------------------
# 2. FULL-DATA target encoding (deliberate leak)
# ----------------------------------------------

means = df.groupby("cat")["y"].mean()
df["te_full"] = df["cat"].map(means).fillna(global_mean)

X_full = df[["te_full"]].values
reg = LinearRegression()

cv = KFold(n_splits=5, shuffle=True, random_state=0)
r2_full = cross_val_score(reg, X_full, y, cv=cv, scoring="r2")

print("R² using FULL-DATA target encoding (inflated due to leakage):")
print(r2_full)
print("Mean R²:", r2_full.mean())

# --------------------------------------------
# 3. Evaluate on a true independent test split
# --------------------------------------------

X_train, X_test, y_train, y_test = train_test_split(
    df[["cat"]], y, test_size=0.4, random_state=0
)

# Compute TE only from the training split
train_means = (
    X_train.join(pd.Series(y_train, name="y"))
            .groupby("cat")["y"].mean()
)

fold_global_mean = y_train.mean()

# Apply to both splits
X_train_enc = X_train["cat"].map(train_means).fillna(fold_global_mean)
X_test_enc  = X_test["cat"].map(train_means).fillna(fold_global_mean)

reg.fit(X_train_enc.values.reshape(-1,1), y_train)
pred_test = reg.predict(X_test_enc.values.reshape(-1,1))

print("\nTrue Test R² (should be ≈ 0):")
print(r2_score(y_test, pred_test))

# ------------------------------------------------------------
# 4. Correct CV-safe target encoding (fold-by-fold construction)
# ------------------------------------------------------------

te_cv = np.zeros(N)
kf = KFold(n_splits=5, shuffle=True, random_state=0)

for train_idx, valid_idx in kf.split(df):
    train_fold = df.iloc[train_idx]
    valid_fold = df.iloc[valid_idx]

    fold_means = train_fold.groupby("cat")["y"].mean()
    fold_global = train_fold["y"].mean()

    enc_vals = valid_fold["cat"].map(fold_means).fillna(fold_global)
    te_cv[valid_idx] = enc_vals

X_te_cv = te_cv.reshape(-1,1)
r2_cvsafe = cross_val_score(reg, X_te_cv, y, cv=5, scoring="r2")

print("\nR² using CV-safe target encoding (correct, no leakage):")
print(r2_cvsafe)
print("Mean R²:", r2_cvsafe.mean())

Using a pipeline

import numpy as np
import pandas as pd
from sklearn.base import BaseEstimator, TransformerMixin
from sklearn.model_selection import KFold, cross_val_score, train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.metrics import r2_score
from sklearn.pipeline import Pipeline

np.random.seed(0)

# ---------------------------------------
# Custom Target Encoder (safe for sklearn)
# ---------------------------------------

class TargetEncoder(BaseEstimator, TransformerMixin):
    """
    Simple target encoder for regression:
    - fit computes category -> mean(target)
    - transform maps categories and fills unseen with global mean
    """
    def __init__(self):
        pass

    def fit(self, X, y):
        X = X.copy()
        X["y"] = y
        self.global_mean_ = y.mean()
        self.cat_means_ = X.groupby("cat")["y"].mean()
        return self

    def transform(self, X):
        X = X.copy()
        out = X["cat"].map(self.cat_means_)
        return out.fillna(self.global_mean_).to_frame()


# -------------------------
# Create synthetic dataset
# -------------------------

n_categories = 200
samples_per_cat = 3
N = n_categories * samples_per_cat

categories = np.repeat(np.arange(n_categories), samples_per_cat)
np.random.shuffle(categories)

# Pure noise regression target
y = np.random.normal(0, 1, size=N)

df = pd.DataFrame({"cat": categories, "y": y})
global_mean = df["y"].mean()

# -------------------------------------------
# 1. BAD: Full-data target encoding (leakage)
# -------------------------------------------

means_full = df.groupby("cat")["y"].mean()
df["te_full"] = df["cat"].map(means_full).fillna(global_mean)

X_full = df[["te_full"]]
reg = LinearRegression()

kf = KFold(n_splits=5, shuffle=True, random_state=0)
r2_full_data = cross_val_score(reg, X_full, y, cv=kf, scoring="r2")

print("R² using FULL-DATA target encoding (leakage):")
print(r2_full_data)
print("Mean:", r2_full_data.mean())

# -----------------------------------------------------
# 2. GOOD: Pipeline with correct target encoding (safe)
# -----------------------------------------------------

pipeline = Pipeline([
    ("te", TargetEncoder()),
    ("reg", LinearRegression())
])

r2_pipeline_safe = cross_val_score(pipeline, df[["cat"]], y, cv=kf, scoring="r2")

print("\nR² using sklearn Pipeline target encoding (no leakage):")
print(r2_pipeline_safe)
print("Mean:", r2_pipeline_safe.mean())

# ---------------------------------------
# 3. True hold-out test set performance
# ---------------------------------------

X_train, X_test, y_train, y_test = train_test_split(
    df[["cat"]], y, test_size=0.4, random_state=0
)

pipeline.fit(X_train, y_train)
pred_test = pipeline.predict(X_test)

print("\nTrue Test R² (should be ≈ 0):")
print(r2_score(y_test, pred_test))
