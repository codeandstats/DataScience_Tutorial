import pandas as pd
import seaborn as sns
from patsy import dmatrices, build_design_matrices
from sklearn.linear_model import LinearRegression
from sklearn.model_selection import train_test_split

# Load Titanic data
df = sns.load_dataset("titanic").dropna()

# Formula
formula1 = """
survived ~ C(sex)*C(pclass) + C(embarked) + C(alone)
           + age + fare + sibsp + parch +
           + I(age**2) + I(fare**2)
"""

formula2 = """
survived ~ C(sex)*C(pclass) + C(embarked) + C(alone)
           + age + fare + sibsp + parch + pclass
           + I(age**2) + I(fare**2)
           + (age + fare + sibsp + parch + pclass)**2
"""
#do NOT follow this order !! leads to data leakage!!
y, X = dmatrices(formula1, data=df, return_type="dataframe")
train_X, test_X, train_y, test_y = train_test_split(X,y, test_size=0.2, random_state=123)

# Split
train_df, test_df = train_test_split(df, test_size=0.2, random_state=123)

# Train: learn + transform
y_train, X_train = dmatrices(formula1, data=train_df, return_type="dataframe")
